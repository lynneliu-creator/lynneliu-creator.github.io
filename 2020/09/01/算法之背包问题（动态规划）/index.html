//优化+标记

void Knapsack()
{
int i,j,k;
for (i = 1; i <= n; i++)
{
for (j = 1; j <= W; j++)
{
for (k = 1; k <= V; k++)
{
if ((goods[i].weight <= j) && (goods[i].c <= k))
{
if (m[i - 1][j][k] > (m[i - 1][j - goods[i].weight][k - goods[i].c] + goods[i].value))
{
m[i][j][k] = m[i - 1][j][k];
s[i][j][k] = 0;
}
else
{
m[i][j][k]=m[i - 1][j - goods[i].weight][k - goods[i].c] + goods[i].value;
s[i][j][k] = 1;
}
}
else
{
m[i][j][k] = m[i - 1][j][k];
s[i][j][k] = 0;
}
}
}
}
}
